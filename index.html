  <p id="devTime">Development %: --</p>

  <p id="ror">RoR: --</p>

<canvas id="chart"></canvas>


<script>

let data = [];

let seconds = 0;

let timer;

let listening = false;


let events = {

  first: null,

  second: null,

  yellow: null

};


let ctx = document.getElementById('chart').getContext('2d');


let chart = new Chart(ctx, {

  type: 'line',

  data: {

    labels: [],

    datasets: [

      { label: 'Temp', data: [], tension: 0.2 },

      { label: 'RoR', data: [], tension: 0.2 },

      { label: 'Events', data: [], showLine: false, pointRadius: 6 }

    ]

  }

});


function startRoast() {

  data = [];

  seconds = 0;

  events = { first:null, second:null, yellow:null };


  chart.data.labels = [];

  chart.data.datasets.forEach(d => d.data = []);

  chart.update();


  timer = setInterval(() => {

    seconds += 30;

    ask();

  }, 30000);


  ask();

}


function stopRoast() {

  clearInterval(timer);

  speak("Roast finished");

}


function ask() {

  if (listening) return;

  listening = true;


  speak("Temperature?");

  

  listen(text => {

    let t = text.toLowerCase();


    detectEvent(t);


    let temp = parseNumber(t);

    if (temp !== null) logTemp(temp);


    listening = false;

  });

}


function detectEvent(text) {

  if (text.includes("first crack")) confirmEvent("first");

  if (text.includes("second crack")) confirmEvent("second");

  if (text.includes("yellow")) confirmEvent("yellow");

}


function confirmEvent(type) {

  speak("Confirm " + type + "?");

  

  listen(resp => {

    if (resp.toLowerCase().includes("yes")) {

      events[type] = seconds;


      let lastTemp = data[data.length-1]?.temp || 0;


      chart.data.datasets[2].data.push({

        x: seconds + "s",

        y: lastTemp

      });


      chart.update();

    }

  });

}


function logTemp(temp) {

  data.push({ time: seconds, temp });


  chart.data.labels.push(seconds + "s");

  chart.data.datasets[0].data.push(temp);


  updateRoR();

  updateDevTime();


  chart.update();

}


function updateRoR() {

  if (data.length < 2) return;


  let last = data[data.length-1];

  let prev = data[data.length-2];


  let ror = (last.temp - prev.temp) / 0.5; // per minute


  chart.data.datasets[1].data.push(ror);


  document.getElementById("ror").innerText =

    "RoR: " + ror.toFixed(1);

}


function updateDevTime() {

  if (!events.first) return;


  let total = seconds;

  let dev = total - events.first;


  let percent = (dev / total) * 100;


  document.getElementById("devTime").innerText =

    "Development %: " + percent.toFixed(1);

}


function submitManual() {

  let val = parseInt(document.getElementById("manualTemp").value);

  if (!isNaN(val)) logTemp(val);

}


function downloadCSV() {

  let csv = "Time,Temp\n";

  data.forEach(d => {

    csv += `${d.time},${d.temp}\n`;

  });


  let blob = new Blob([csv], { type: 'text/csv' });

  let a = document.createElement("a");

  a.href = URL.createObjectURL(blob);

  a.download = "roast.csv";

  a.click();

}


function listen(cb) {

  const rec = new (window.SpeechRecognition || window.webkitSpeechRecognition)();

  rec.onresult = e => cb(e.results[0][0].transcript);

  rec.start();

}


function speak(text) {

  speechSynthesis.speak(new SpeechSynthesisUtterance(text));

}


function parseNumber(text) {

  if (text.match(/\d+/)) return parseInt(text.match(/\d+/)[0]);


  const map = {

    one:1,two:2,three:3,four:4,five:5,six:6,seven:7,eight:8,nine:9,

    ten:10,twenty:20,thirty:30,forty:40,fifty:50,

    sixty:60,seventy:70,eighty:80,ninety:90,hundred:100

  };


  let total = 0;

  text.split(" ").forEach(w => {

    if (map[w]) {

      if (map[w] === 100) total *= 100;

      else total += map[w];

    }

  });


  return total || null;

}
